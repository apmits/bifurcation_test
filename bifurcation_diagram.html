# define the logistic map    https://en.wikipedia.org/wiki/Logistic_map
def logistic(r, x):
    #        /\_ r is growth rate
    #            x is population
    return r * x * (1 - x)

# Array of r values, for the x axis of the bifurcation plot
_r = []
# Array of x(t) values, for the y axis of the bifurcation plot
_x = []

for r in r_range:
    _r.append(r)
    # m = 0.1
    #   Start with a random value of m instead of remaining stuck
    #   on a particular branch of the diagram
    m = np.random.random()  # Return random floats in the half-open interval [0.0, 1.0).
    for n in range(1500):
      m = logistic(r, m)
      # m = (r*m)*(1-m)
    # Collection of data in Y must be done once per value of r
    _x.append(m)
